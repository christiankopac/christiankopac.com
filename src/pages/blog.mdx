export const pageQuery = graphql`
  query {
    allMdx(
      sort: { frontmatter: { date: DESC } }
      filter: {
        frontmatter: { status: { eq: "published" } }
        internal: { contentFilePath: { regex: "/blog/" } }
      }
    ) {
      nodes {
        frontmatter {
          title
          date
          slug
          tags
          excerpt
        }
      }
    }
  }
`
